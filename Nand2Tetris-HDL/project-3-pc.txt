CHIP PC {
    IN in[16], reset, load, inc;
    OUT out[16];
    
    PARTS:
    Inc16(in=regOut , out=increment );
    Mux16(a=regOut , b=increment , sel=inc , out=incOut );
    Mux16(a=incOut , b=in , sel=load , out=loadOut );
    Mux16(a=loadOut , b[0..15]=false , sel=reset , out=preOut );
    Register(in=preOut , load=true , out=out, out=regOut );
}
